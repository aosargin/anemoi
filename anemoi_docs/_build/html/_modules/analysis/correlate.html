def compare_df_columns(cols_1, cols_2):
    return Counter(cols_1) == Counter(cols_2)

def compare_sorted_df_columns(cols_1, cols_2):
    return sorted(cols_1) == sorted(cols_2)

def valid_correlation(df, ref, site):
    # Assume df.columns == ['Ref', 'Site', 'Dir']
    df = df.loc[:, [ref,site]].dropna()
    
    if df.shape[0] < 6:
        warnings.warn("Warning: you are trying to correalate between less than six points.")
        return False

    if ref == site:
        return False

    return True

[docs]def calculate_R2(df, ref='RefWS', site='SiteWS'):
    '''Return a single R2 between two wind speed columns'''

    df = df.loc[:, [ref,site]].dropna()
    
    if not valid_correlation(df=df, ref=ref, site=site):
        return np.nan
    
    return df.corr().iloc[0,1]**2



[docs]def calculate_IEC_uncertainty(df, ref='RefWS', site='SiteWS'):
    '''Calculate the IEC correlation uncertainty between two wind speed columns'''
    
    df = df.loc[:, [ref,site]].dropna()
    
    if not valid_correlation(df=df, ref=ref, site=site):
        return np.nan

    X = df[ref].values
    Y = df[site].values
    uncert = np.std(Y/X)*100/len(X)
    return uncert

    

[docs]def calculate_EDF_uncertainty(df, ref='RefWS', site='SiteWS'):
    '''Calculate the EDF estimated correaltion uncetianty between two wind speed columns. 
    Assumes a correalation forced through the origin'''
    
    df = df.loc[:, [ref,site]].dropna()
    
    if not valid_correlation(df=df, ref=ref, site=site):
        return np.nan

    Sxx = np.sum(df[ref]**2)
    Syy = np.sum(df[site]**2)
    Sxy = np.sum(df[ref]*df[site])
    B = 0.5*(Sxx - Syy)/Sxy
    SU = -B + np.sqrt(B**2 + 1)

    e2 = np.sum((df[site] - SU*df[ref])**2)/(1 + SU**2)
    Xsi2 = e2/(df.shape[0] - 1)
    uncert = np.sqrt((Xsi2*SU**2)*(Sxx*Sxy**2 + 0.25*((Sxx - Syy)**2)*Sxx)/((B**2 + 1.0)*Sxy**4))
    return uncert



[docs]def correlate_least_squares(df, ref='RefWS', site='SiteWS', force_through_origin=True):    
    '''Calculate the slope and offset between two wind speed columns using ordinary least squares regression.
    
    :Parameters: 
    
    wind speed data: DataFrame
        Measured wind speed data from MetMast.data

    ref: string, default 'RefWS'
        Label of the reference wind speed data column 

    site: string, default 'SiteWS'
        Label of the site wind speed data column 

    force_through_origin: boolean, default True
        Force the correlation through the origin (offset equal to zero)
    
    :Returns:

    out: slope, offset, R2, uncert, points
    
    '''
    df = df.loc[:, [ref,site]].dropna()
    points = df.shape[0]
    
    if not valid_correlation(df=df, ref=ref, site=site):
        slope = np.nan 
        offset = np.nan 
        R2 = np.nan 
        uncert = np.nan
        return slope, offset, R2, uncert, points
    
    R2 = calculate_R2(df=df, ref=ref, site=site)
    uncert = calculate_IEC_uncertainty(df=df, ref=ref, site=site)
    
    if force_through_origin:
        df.loc[:,'offset'] = 0
    else:
        df.loc[:,'offset'] = 1
    
    X = df.loc[:, [ref,'offset']].values
    Y = df[site].values
    slope, offset = np.linalg.lstsq(X, Y)[0]
    return slope, offset, R2, uncert, points


def f_with_offset(B, x):
    return B[0]*x + B[1]

def f_without_offset(B, x):
    return B[0]*x

def correlate_orthoginal_distance(df, ref='RefWS', site='SiteWS', force_through_origin=True):
    df = df.loc[:, [ref,site]].dropna()
    points = df.shape[0]
    
    if not valid_correlation(df=df, ref=ref, site=site):
        slope = np.nan 
        offset = np.nan 
        R2 = np.nan 
        uncert = np.nan
        return slope, offset, R2, uncert, points
       
    R2 = calculate_R2(df=df, ref=ref, site=site)
    uncert = calculate_IEC_uncertainty(df=df, ref=ref, site=site)
    
    X = df[ref].values
    Y = df[site].values

    realdata = odrpack.RealData(X, Y)

    if force_through_origin:
        linear = odrpack.Model(f_without_offset)
        odr = odrpack.ODR(realdata, linear, beta0=[1.0])
        slope = odr.run().beta[0]
        offset = 0
    else:
        linear = odrpack.Model(f_with_offset)
        odr = odrpack.ODR(realdata, linear, beta0=[1.0, 0.0])
        slope, offset = odr.run().beta[0], odr.run().beta[1]
    
    return slope, offset, R2, uncert, points
            
def correlate_principal_component(df, ref='RefWS', site='SiteWS'):
    df = df.loc[:, [ref,site]].dropna()
    points = df.shape[0]
    
    if not valid_correlation(df=df, ref=ref, site=site):
        slope = np.nan 
        offset = np.nan 
        R2 = np.nan 
        uncert = np.nan
        return slope, offset, R2, uncert, points
       
    R2 = calculate_R2(df=df, ref=ref, site=site)
    uncert = calculate_IEC_uncertainty(df=df, ref=ref, site=site)
    
    X = df[ref].values
    Y = df[site].values
    
    beta = 0.5*(np.sum(X**2)-np.sum(Y**2)) / np.sum(X*Y)
    slope = -beta + np.sqrt(beta**2 + 1)
    offset = 0
    return slope, offset, R2, uncert, points

def correlate_robust_linear_model(df, ref='RefWS', site='SiteWS', force_through_origin=True):
    df = df.loc[:, [ref,site]].dropna()
    points = df.shape[0]
    
    if not valid_correlation(df=df, ref=ref, site=site):
        slope = np.nan 
        offset = np.nan 
        R2 = np.nan 
        uncert = np.nan
        return slope, offset, R2, uncert, points
       
    R2 = calculate_R2(df=df, ref=ref, site=site)
    uncert = calculate_IEC_uncertainty(df=df, ref=ref, site=site)
    
    X = df[ref].values
    Y = df[site].values
    
    if not force_through_origin:
        X = sm.add_constant(X)
    else:
        X = [np.zeros(X.shape[0]), X]
        X = np.column_stack(X)
    
    mod = sm.RLM(Y, X)
    resrlm = mod.fit()
    offset, slope = resrlm.params
    R2 = sm.WLS(mod.endog, mod.exog, weights=mod.fit().weights).fit().rsquared
    return slope, offset, R2, uncert, points

def append_dir_bin(df, dir_column, sectors=12, drop_dir_column=True):
    bin_width = 360.0/sectors
    df['dir_bin'] = np.floor_divide(np.mod(df[dir_column] + (bin_width/2.0),360),bin_width)+1
    df['dir_bin'] = df['dir_bin'].astype(int)
    if drop_dir_column:
        df.drop(dir_column, axis=1, inplace=True)
    return df

def correlate_ws_binned_by_direction(df, ref='RefWS', site='SiteWS', dir_column='Dir', sectors=12, method='ODR', force_through_origin=False):
    # Method: defaults 'ODR', 'OLS', 'PCA', 'RLM'
    df = df.loc[:, [ref,site,dir_column]].dropna()
    dir_bins = np.arange(1,sectors+1)
    results = pd.DataFrame(index=dir_bins, columns=['slope', 'offset', 'R2', 'uncert', 'points'])
    results.index.name = 'dir_bin'

    if not valid_correlation(df=df, ref=ref, site=site):
        return results

    df = append_dir_bin(df, dir_column, sectors=sectors, drop_dir_column=True)
    
    for dir_bin in dir_bins:
        dir_bin_df = df.loc[df['dir_bin']==dir_bin, [ref,site]]
        points = dir_bin_df.shape[0]
        
        if not valid_correlation(df=dir_bin_df, ref=ref, site=site):
            invalid_results = np.array([np.nan, np.nan, np.nan, np.nan, points])
            results.loc[dir_bin, ['slope', 'offset', 'R2' , 'uncert', 'points']] = invalid_results

        else:
            uncert = calculate_IEC_uncertainty(df=dir_bin_df, ref=ref, site=site)
    
            if (method == 'ODR') and (force_through_origin is True):
                slope, offset, R2, uncert, points = correlate_orthoginal_distance(df=dir_bin_df, ref=ref, site=site, force_through_origin=force_through_origin)
            elif (method == 'ODR') and (force_through_origin is False):
                slope, offset, R2, uncert, points = correlate_orthoginal_distance(df=dir_bin_df, ref=ref, site=site, force_through_origin=force_through_origin)
            elif (method == 'OLS') and (force_through_origin is True): 
                slope, offset, R2, uncert, points = correlate_least_squares(df=dir_bin_df, ref=ref, site=site, force_through_origin=force_through_origin)
            elif (method == 'OLS') and (force_through_origin is False):
                slope, offset, R2, uncert, points = correlate_least_squares(df=dir_bin_df, ref=ref, site=site, force_through_origin=force_through_origin)
            elif (method == 'PCA') and (force_through_origin is True): 
                slope, offset, R2, uncert, points = correlate_principal_component(df=dir_bin_df, ref=ref, site=site)
            elif (method == 'PCA') and (force_through_origin is False):
                slope, offset, R2, uncert, points = correlate_orthoginal_distance(df=dir_bin_df, ref=ref, site=site, force_through_origin=force_through_origin)
            
            valid_results = np.array([slope, offset, R2, uncert, points])
            results.loc[dir_bin, ['slope', 'offset', 'R2' , 'uncert', 'points']] = valid_results

    return results

def correlate_masts_10_minute(ref_mast, site_mast, ref_ws_sensor=None, site_ws_sensor=None, force_through_origin=False):
    #Return slope, offset, R2, uncertainty, data
    ref_mast.check_mast_data_size()
    site_mast.check_mast_data_size()

    ref_ws_sensor = ref_mast.check_and_return_mast_ws_sensor(ref_ws_sensor)
    site_ws_sensor = site_mast.check_and_return_mast_ws_sensor(site_ws_sensor)

    ref_data = ref_mast.return_sensor_data([ref_ws_sensor])
    site_data = site_mast.return_sensor_data([site_ws_sensor])
    data_10_min = pd.concat([ref_data, site_data], axis=1, join='inner').dropna()
    data_10_min.columns = ['RefWS', 'SiteWS']
    
    results_index = masts = pd.MultiIndex.from_tuples([(site_mast.name, ref_mast.name)], names=['Site', 'Ref'])
    results = pd.DataFrame(index=results_index, columns=['slope', 'offset', 'R2', 'uncert', 'points'])
    valid_results = correlate_orthoginal_distance(data_10_min, ref='RefWS', site='SiteWS', force_through_origin=force_through_origin)
    results.loc[pd.IndexSlice[site_mast.name, ref_mast.name], ['slope', 'offset', 'R2' , 'uncert', 'points']] = valid_results
    return results

def correlate_masts_10_minute_by_direction(ref_mast, site_mast, ref_ws_sensor=None, ref_dir_sensor=None, site_ws_sensor=None, force_through_origin=False, sectors=12):
    #Return correaltion results: slope, offset, R2, uncertainty
    ref_mast.check_mast_data_size()
    site_mast.check_mast_data_size()

    ref_ws_sensor = ref_mast.check_and_return_mast_ws_sensor(ref_ws_sensor)
    site_ws_sensor = site_mast.check_and_return_mast_ws_sensor(site_ws_sensor)
    ref_dir_sensor = ref_mast.check_and_return_mast_dir_sensor(ref_dir_sensor)

    ref_data = ref_mast.return_sensor_data([ref_ws_sensor, ref_dir_sensor])
    site_data = site_mast.return_sensor_data(site_ws_sensor)
    df = pd.concat([ref_data, site_data], axis=1, join='inner', keys=['Ref', 'Site']).dropna()
    df.columns = ['RefWS', 'RefDir', 'SiteWS']

    results = correlate_ws_binned_by_direction(df=df, ref='RefWS', site='SiteWS', dir_column='RefDir', sectors=sectors, method='ODR', force_through_origin=force_through_origin)
    results = results.reset_index()
    results['Ref'] = ref_mast.name
    results['Site'] = site_mast.name
    results = results.set_index(['Site', 'Ref', 'dir_bin'])
    return results

def correlate_masts_daily(ref_mast, site_mast, ref_ws_sensor=None, site_ws_sensor=None, minimum_recovery_rate=0.7):
    #Return slope, offset, R2, uncertainty, data
    ref_mast.check_mast_data_size()
    site_mast.check_mast_data_size()
    
    ref_ws_sensor = ref_mast.check_and_return_mast_ws_sensor(ref_ws_sensor)
    site_ws_sensor = site_mast.check_and_return_mast_ws_sensor(site_ws_sensor)

    if minimum_recovery_rate > 1:
        minimum_recovery_rate = minimum_recovery_rate/100.0

    ref_data_daily_mean = ref_mast.resample_sensor_data(ref_ws_sensor, freq='daily', minimum_recovery_rate=minimum_recovery_rate)
    site_data_daily_mean = site_mast.resample_sensor_data(site_ws_sensor, freq='daily', minimum_recovery_rate=minimum_recovery_rate)
    data_daily = pd.concat([ref_data_daily_mean, site_data_daily_mean], axis=1, keys=['Ref', 'Site']).dropna()
    data_daily.columns  = ['RefWS', 'SiteWS']

    if data_daily.shape[0] < 7:
        raise ValueError('Less than one week of valid overlapping data.')
    
    correlation_results = pd.DataFrame(index='corr_daily', columns=['slope', 'offset', 'R2', 'uncert', 'points'])
    
    slope, offset = correlate_orthoginal_distance(data_daily, ref='RefWS', site='SiteWS', force_through_origin=False)
    uncert = calculate_IEC_uncertainty(data_daily, ref='RefWS', site='SiteWS')
    R2 = calculate_R2(data_daily, ref='RefWS', site='SiteWS')
    
    correlation_results.loc['corr_daily', ['slope', 'offset', 'R2', 'uncert', 'points']] = [slope, offset, R2, uncert, data_daily.shape[0]]
    return correlation_results, data_daily

def correlate_masts_daily_by_month(ref_mast, site_mast, ref_ws_sensor=None, site_ws_sensor=None, minimum_recovery_rate=0.7):
    #Return slope, offset, R2, uncertainty, data
    ref_mast.check_mast_data_size()
    site_mast.check_mast_data_size()
    
    ref_ws_sensor = ref_mast.check_and_return_mast_ws_sensor(ref_ws_sensor)
    site_ws_sensor = site_mast.check_and_return_mast_ws_sensor(site_ws_sensor)

    if minimum_recovery_rate > 1:
        minimum_recovery_rate = minimum_recovery_rate/100.0

    ref_data_daily_mean = ref_mast.resample_sensor_data(ref_ws_sensor, freq='daily', minimum_recovery_rate=minimum_recovery_rate)
    site_data_daily_mean = site_mast.resample_sensor_data(site_ws_sensor, freq='daily', minimum_recovery_rate=minimum_recovery_rate)
    data_daily = pd.concat([ref_data_daily_mean, site_data_daily_mean], axis=1, keys=['Ref', 'Site']).dropna()
    data_daily.columns  = ['RefWS', 'SiteWS']

    if data_daily.shape[0] < 7:
        raise ValueError('Less than one week of valid overlapping data.')
    
    correlation_results = pd.DataFrame(index=np.arange(1,13), columns=['slope', 'offset', 'R2', 'uncert', 'points'])
    correlation_results.index.name = 'month'

    for month in correlation_results.index:
        data_daily_month_filtered = data_daily.loc[data_daily.index.month == month, :]
        
        if data_daily_month_filtered.shape[0] > 7:
            # raise ValueError('Less than one week of valid overlapping data in month: {}.'.format(month))
            slope, offset = correlate_orthoginal_distance(data_daily_month_filtered, ref='RefWS', site='SiteWS', force_through_origin=False)
            uncert = calculate_IEC_uncertainty(data_daily_month_filtered, ref='RefWS', site='SiteWS')
            R2 = calculate_R2(data_daily_month_filtered, ref='RefWS', site='SiteWS')
        
            correlation_results.loc[month, ['slope', 'offset', 'R2', 'uncert', 'points']] = [slope, offset, R2, uncert, data_daily.shape[0]]
    correlation_results.loc[:, ['slope', 'offset', 'R2']] = correlation_results.loc[:, ['slope', 'offset', 'R2']].fillna(correlation_results.loc[:, ['slope', 'offset', 'R2']].mean())
    correlation_results.points.fillna(0, inplace=True)
    return correlation_results, data_daily

def plot_wind_speed_correlation(df, ref='RefWS', site='SiteWS', slope=None, R2=None, uncert=None, title='Wind speed correlation'):
    df.columns = df.columns.get_level_values(level='Sensors')
    df = df.loc[:, [ref,site]].dropna()

    if (df.shape[0] != 0) and (ref != site):
        fig = plt.figure(figsize=[6,6])
        ax = fig.add_subplot(111)
        df.plot(kind='scatter', x=ref, y=site, xlim=[0,30], ylim=[0,30], ax=ax)
        if slope is not None:
            ax.plot([0,30], [0,30*slope], '-', color='red')
        if R2 is not None:
            ax.set_title(title + ' (R2: {:.3f})'.format(R2))
        ax.set_xlabel(ref)
        ax.set_ylabel(site)
        plt.close()
        return fig
